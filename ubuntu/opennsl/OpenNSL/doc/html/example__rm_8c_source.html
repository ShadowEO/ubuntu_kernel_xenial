91 int example_l2_params_get(uint16 *vlan, int *port, opennsl_mac_t mac)
92 {
93  int val;
94  char macstr[18];
95 
96  printf("Enter VLAN: \n");
97  if(example_read_user_choice(&val) != OPENNSL_E_NONE)
98  {
99  printf("Invalid value entered. Please re-enter.\n");
100  return OPENNSL_E_FAIL;
101  }
102  *vlan = (uint16) val;
103 
104  printf("Enter Port Number: \n");
105  if(example_read_user_choice(&val) != OPENNSL_E_NONE)
106  {
107  printf("Invalid value entered. Please re-enter.\n");
108  return OPENNSL_E_FAIL;
109  }
110  *port = val;
111 
112  printf("Enter MAC address in xx:xx:xx:xx:xx:xx format: \n");
113  if(example_read_user_string(macstr, sizeof(macstr)) == NULL)
114  {
115  printf("\n\rInvalid MAC address entered.\n\r");
116  return OPENNSL_E_FAIL;
117  }
118 
119  if(opennsl_mac_parse(macstr, mac) != OPENNSL_E_NONE)
120  {
121  printf("\n\rFailed to parse input MAC address.\n\r");
122  return OPENNSL_E_FAIL;
123  }
124  return OPENNSL_E_NONE;
125 }

135 void example_rm_show_profile(opennsl_rm_profile_t *profile)
136 {
137  int jj;
138 
139  for(jj=0; jj < RM_MAX_UNITS;jj++)
140  {
141  printf("Unit:%d Ports = ",jj);
142  printf(EXAMPLE_RM_PORT_PBMP_FORMAT,
143  EXAMPLE_RM_PORT_PBMP_SHOW(profile->port_bitmap[jj]));
144  printf("\n");
145  }
146 
147  printf("\nVLANs : ");
148  for(jj = (_SHR_BITDCLSIZE(OPENNSL_VLAN_COUNT) - 1);jj >= 0;jj--)
149  {
150  printf("%08X ",profile->vlan_bitmap[jj]);
151  }
152  printf("\n");
153 
154  return;
155 }

165 int main(int argc, char *argv[])
166 {
167  int choice;
168  int rval;
169  opennsl_error_t rv;
170  opennsl_vlan_t vlan = 1;
171  opennsl_port_t port;
172  opennsl_mac_t mac;
173  opennsl_l2_addr_t addr;
174  int mod_id = 0;
175  int unit = DEFAULT_UNIT;
176  void *user_data = NULL;
177  int index = 0;
178  opennsl_pbmp_t port_bitmap;
179  opennsl_vlan_vector_t vlan_bitmap;
180  opennsl_rm_profile_t profile;
181 
182  if(strcmp(argv[0], "gdb") == 0)
183  {
184  index = 1;
185  }
186 
187  if((argc != (index + 1)) || ((argc > (index + 1))
188  && (strcmp(argv[index + 1], "--help") == 0))) {
189  printf("%s\n\r", example_usage);
190  return OPENNSL_E_PARAM;
191  }
192 
193  /* OpenNSL initialization is needed for RM initialization to succeed. */
194  printf("Initializing the system.This will also initialize RM\r\n");
195  rv = opennsl_driver_init((opennsl_init_t *) NULL);
196  if(rv != OPENNSL_E_NONE) {
197  printf("\r\nFailed to initialize the system. Error %s\r\n",
198  opennsl_errmsg(rv));
199  return rv;
200  }
201 
202  /* cold boot initialization commands */
203  rv = example_port_default_config(unit);
204  if (rv != OPENNSL_E_NONE) {
205  printf("\r\nFailed to apply default config on ports, rc = %d (%s).\r\n",
206  rv, opennsl_errmsg(rv));
207  }
208 
209  while (1) {
210  printf("\r\nUser menu: Select one of the following options\r\n");
211  printf(" 1. Show the NOS Profile values\n");
212  printf(" 2. Add L2 entry\n");
213  printf(" 3. Remove L2 entry\n");
214  printf(" 4. Show L2 table\n");
215  printf(" 5. Update the Ports of the default client\n");
216  printf(" 6. Update the VLANs of the default client\n");
217 #ifdef INCLUDE_DIAG_SHELL
218  printf(" 9. Launch diagnostic shell\n");
219 #endif
220  printf(" 0. Quit the application\n");
221 
222  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
223  {
224  printf("Invalid option entered. Please re-enter.\n");
225  continue;
226  }
227 
228  switch(choice){
229  case 1:
230  {
231  memset(&profile,0,sizeof(opennsl_rm_profile_t));
232  rval = opennsl_rm_profile_get(DEFAULT_RM_CLIENTID,&profile);
233  if (rval == OPENNSL_RM_E_NONE)
234  {
235  printf("RM profile get is successful\n");
236  example_rm_show_profile(&profile);
237  }
238  else
239  {
240  printf("RM profile get failed:%d\n",rval);
241  }
242 
243  break;
244  } /* End of case 1 */
245 
246  case 2:
247  {
248  if(example_l2_params_get(&vlan, &port, mac) != OPENNSL_E_NONE)
249  {
250  continue;
251  }
252 
253  opennsl_l2_addr_t_init(&addr, mac, vlan);
254 
255  OPENNSL_GPORT_MODPORT_SET(addr.port, mod_id, port);
256  addr.flags = OPENNSL_L2_STATIC;
257 
258  rv = opennsl_l2_addr_add(unit, &addr);
259  if (rv == OPENNSL_E_NONE)
260  {
261  printf("L2 static entry added successfully\n");
262  printf(" VLAN : %d\n", addr.vid);
263  printf(" Gport : 0x%0x\n", addr.port);
264  printf(" MAC : %02x:%02x:%02x:%02x:%02x:%02x\n",
265  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
266  }
267  else
268  {
269  printf("Failed to add L2 static entry. Error %s\n",
270  opennsl_errmsg(rv));
271  }
272  break;
273  } /* End of case 2 */
274 
275  case 3:
276  {
277  if(example_l2_params_get(&vlan, &port, mac) != OPENNSL_E_NONE)
278  {
279  continue;
280  }
281 
282  /* Check whether the station exists in the firewall rules */
283  rv = opennsl_l2_addr_get(unit, mac, vlan, &addr);
284 
285  if (rv != OPENNSL_E_NONE)
286  {
287  printf("\n\rFailed to get L2 entry with the input parameters. "
288  "Error %s\n\r", opennsl_errmsg(rv));
289  continue;
290  }
291 
292  OPENNSL_GPORT_MODPORT_SET(addr.port, mod_id, port);
293  addr.flags = OPENNSL_L2_STATIC;
294 
295  rv = opennsl_l2_addr_delete(unit, mac, vlan);
296  if (rv == OPENNSL_E_NONE)
297  {
298  printf("L2 static entry deleted successfully\n");
299  printf(" VLAN : %d\n", addr.vid);
300  printf(" Gport : 0x%0x\n", addr.port);
301  printf(" MAC : %02x:%02x:%02x:%02x:%02x:%02x\n",
302  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
303  }
304  else
305  {
306  printf("Failed to delete L2 static entry. Error %s\n",
307  opennsl_errmsg(rv));
308  }
309  break;
310  } /* End of case 3 */
311 
312  case 4:
313  {
314  /* Iterate over all valid entries in the L2 table */
315  rv = opennsl_l2_traverse(unit, &_opennsl_l2_traverse_cb, user_data);
316  if(rv != OPENNSL_E_NONE)
317  {
318  printf("\r\nFailed to iterate over L2 table. rv: %s\r\n",
319  opennsl_errmsg(rv));
320  }
321  break;
322  } /* End of case 4 */
323 
324  case 5:
325  {
326  /* Update the default profile with ports - 21-40 */
327  OPENNSL_PBMP_CLEAR(port_bitmap);
328  for(index = 21; index <= 40; index++)
329  {
330  OPENNSL_PBMP_PORT_ADD(port_bitmap, index);
331  }
332 
333  if(OPENNSL_RM_E_NONE !=
334  opennsl_rm_profile_ports_update(DEFAULT_RM_CLIENTID, unit, port_bitmap))
335  {
336  printf("RM profile update failed.\n");
337  }
338  else
339  {
340  printf("RM profile successfully updated.\n");
341  }
342  break;
343  } /* End of case 5 */
344 
345  case 6:
346  {
347  /* Update the default profile with VLANS - 101-200 */
348  OPENNSL_VLAN_VEC_ZERO(vlan_bitmap);
349  for(index = 101; index <= 200; index++)
350  {
351  OPENNSL_VLAN_VEC_SET(vlan_bitmap, index);
352  }
353 
354  if(OPENNSL_RM_E_NONE !=
355  opennsl_rm_profile_vlans_update(DEFAULT_RM_CLIENTID, vlan_bitmap))
356  {
357  printf("RM profile update failed.\n");
358  }
359  else
360  {
361  printf("RM profile successfully updated.\n");
362  }
363 
364  break;
365  } /* End of case 6 */
366 
367 #ifdef INCLUDE_DIAG_SHELL
368  case 9:
369  {
370  opennsl_driver_shell();
371  break;
372  }
373 #endif
374 
375  case 0:
376  {
377  printf("Exiting the application.\n");
378  return OPENNSL_E_NONE;
379  }
380  default:
381  break;
382  } /* End of switch */
383  } /* End of while */
384 
385  return OPENNSL_E_NONE;
386 }
