94 int example_bst_default_profile_set (int unit)
95 {
96  int rc;
97  int bid;
98  int depth;
99  int index;
100  int max_threshold;
101  int gport;
102  int port;
103  opennsl_cosq_bst_profile_t profile;
104  opennsl_info_t info;
105  opennsl_port_config_t pcfg;
106 
107  /* Get port configuration */
108  rc = opennsl_port_config_get(unit, &pcfg);
109  if (rc != OPENNSL_E_NONE)
110  {
111  printf("Failed to get port configuration, rc = %d (%s).\n",
112  rc, opennsl_errmsg(rc));
113  return rc;
114  }
115 
116  if((rc = opennsl_info_get(unit, &info)) != OPENNSL_E_NONE)
117  {
118  printf("\r\nFailed to get switch device details, rc = %d (%s).\r\n",
119  rc, opennsl_errmsg(rc));
120  return rc;
121  }
122 
123  for (bid = 0; bid < _BST_STAT_ID_MAX_COUNT; bid++)
124  {
125  if(_BRCM_IS_CHIP_TH(info)) /* Tomahawk */
126  {
127  if (bid == _BST_STAT_ID_UCAST ||
128  bid == _BST_STAT_ID_UCAST_GROUP)
129  {
130  max_threshold = 0xFFF * 208;
131  }
132  else
133  {
134  max_threshold = 0x7FFF * 208;
135  }
136  }
137  else /* Trident 2 */
138  {
139  /* Find out the maximum threshold for a bid */
140  if (bid == _BST_STAT_ID_UCAST ||
141  bid == _BST_STAT_ID_EGR_UCAST_PORT_SHARED ||
142  bid == _BST_STAT_ID_UCAST_GROUP)
143  {
144  max_threshold = 0x3FFF * 208;
145  }
146  else if (bid == _BST_STAT_ID_PRI_GROUP_HEADROOM)
147  {
148  max_threshold = 0xFFF * 208;
149  }
150  else
151  {
152  max_threshold = 0x1FFFF * 208;
153  }
154  }
155 
156  profile.byte = max_threshold;
157 
158  if (bid == _BST_STAT_ID_EGR_POOL ||
159  bid == _BST_STAT_ID_EGR_MCAST_POOL ||
160  bid == _BST_STAT_ID_EGR_UCAST_PORT_SHARED ||
161  bid == _BST_STAT_ID_EGR_PORT_SHARED ||
162  bid == _BST_STAT_ID_UCAST_GROUP ||
163  bid == _BST_STAT_ID_PORT_POOL ||
164  bid == _BST_STAT_ID_ING_POOL)
165  {
166  depth = 4;
167  }
168  else if (bid == _BST_STAT_ID_PRI_GROUP_SHARED ||
169  bid == _BST_STAT_ID_PRI_GROUP_HEADROOM ||
170  bid == _BST_STAT_ID_UCAST ||
171  bid == _BST_STAT_ID_MCAST)
172  {
173  depth = 8;
174  }
175  else if (bid == _BST_STAT_ID_RQE_QUEUE)
176  {
177  depth = 11;
178  }
179  else
180  {
181  depth = 1;
182  }
183 
184  /* Set default BST profiles */
185  if (bid == _BST_STAT_ID_ING_POOL ||
186  bid == _BST_STAT_ID_EGR_MCAST_POOL ||
187  bid == _BST_STAT_ID_EGR_POOL ||
188  bid == _BST_STAT_ID_RQE_QUEUE ||
189  bid == _BST_STAT_ID_UCAST_GROUP ||
190  bid == _BST_STAT_ID_DEVICE)
191  {
192  gport = 0;
193  for (index = 0; index < depth; index++)
194  {
195  rc = opennsl_cosq_bst_profile_set (unit, gport, index, bid, &profile);
196  if (rc != OPENNSL_E_NONE)
197  {
198  printf("\r\nFailed to set the BST profile for gport %d bid %d index %d, rc = %d (%s).\r\n",
199  gport, bid, index, rc, opennsl_errmsg(rc));
200  return rc;
201  }
202  } /* Iter ... Cosq */
203  }
204  else if (bid == _BST_STAT_ID_PRI_GROUP_SHARED ||
205  bid == _BST_STAT_ID_PRI_GROUP_HEADROOM ||
206  bid == _BST_STAT_ID_UCAST ||
207  bid == _BST_STAT_ID_MCAST ||
208  bid == _BST_STAT_ID_PORT_POOL ||
209  bid == _BST_STAT_ID_EGR_PORT_SHARED ||
210  bid == _BST_STAT_ID_EGR_UCAST_PORT_SHARED)
211  {
212  OPENNSL_PBMP_ITER(pcfg.e, port)
213  {
214  /* Get GPORT */
215  rc = opennsl_port_gport_get (unit, port, &gport);
216  if (rc != OPENNSL_E_NONE)
217  {
218  printf("\r\nFailed to get gport for port %d, rc = %d (%s).\r\n",
219  port, rc, opennsl_errmsg(rc));
220  return rc;
221  }
222 
223  for (index = 0; index < depth; index++)
224  {
225  rc = opennsl_cosq_bst_profile_set (unit, gport, index, bid, &profile);
226  if (rc != OPENNSL_E_NONE)
227  {
228  printf("\r\nFailed to set the BST profile for gport %d bid %d index %d, rc = %d (%s).\r\n",
229  gport, bid, index, rc, opennsl_errmsg(rc));
230  return rc;
231  }
232  } /* Iter ... index */
233  } /* Iter ... port */
234  }
235 
236  } /* Iter ... bid */
237 
238  return 0;
239 }

270 /* Main function for BST application */
271 int main(int argc, char *argv[])
272 {
273  opennsl_error_t rc;
274  int choice;
275  int unit = DEFAULT_UNIT;
276  int index,i,j;
277  opennsl_gport_t gport;
278  opennsl_port_t port;
279  opennsl_cos_queue_t cosq;
280  uint32 options = 0;
281  void *user_data = NULL;
282 
283  example_bst_counter_t id_list[MAX_COUNTERS] = {
284  {_BST_STAT_ID_UCAST, "opennslBstStatIdUcast"},
285  {_BST_STAT_ID_MCAST, "opennslBstStatIdMcast"},
286  {_BST_STAT_ID_PRI_GROUP_SHARED, "opennslBstStatIdPriGroupShared"},
287  {_BST_STAT_ID_PRI_GROUP_HEADROOM, "opennslBstStatIdPriGroupHeadroom"}
288  };
289  uint64 val[MAX_COUNTERS][MAX_COSQ_COUNT];
290 
291  /* Intitializing val array */
292  for (i = 0; i < MAX_COUNTERS; i++)
293  {
294  for (j = 0; j < MAX_COSQ_COUNT; j++)
295  {
296  val[i][j] = 0;
297  }
298  }
299 
300  if ((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0)))
301  {
302  printf("%s\n\r", example_usage);
303  return OPENNSL_E_PARAM;
304  }
305 
306  /* Initialize the system. */
307  printf("Initializing the system.\r\n");
308  rc = opennsl_driver_init((opennsl_init_t *) NULL);
309 
310  if (rc != OPENNSL_E_NONE)
311  {
312  printf("\r\nFailed to initialize the system, rc = %d (%s).\r\n",
313  rc, opennsl_errmsg(rc));
314  return rc;
315  }
316 
317  /* Add ports to default vlan. */
318  printf("Adding ports to default vlan.\r\n");
319  rc = example_switch_default_vlan_config(unit);
320  if (rc != OPENNSL_E_NONE)
321  {
322  printf("\r\nFailed to add default ports, rc = %d (%s).\r\n",
323  rc, opennsl_errmsg(rc));
324  return rc;
325  }
326 
327  rc = example_port_default_config(unit);
328  if (rc != OPENNSL_E_NONE)
329  {
330  printf("\r\nFailed to apply default config on ports, rc = %d (%s).\r\n",
331  rc, opennsl_errmsg(rc));
332  }
333 
334  /* Set default BST profiles */
335  rc = example_bst_default_profile_set (unit);
336  if (rc != OPENNSL_E_NONE)
337  {
338  printf("\r\nFailed to set default profile for BST, rc = %d (%s).\r\n",
339  rc, opennsl_errmsg(rc));
340  return rc;
341  }
342  printf("Default BST profiles are set successfully.\r\n");
343 
344  /* Register Hw trigger callback*/
345  rc = opennsl_switch_event_register (unit,
346  example_bst_trigger_callback,
347  user_data);
348  if (rc != OPENNSL_E_NONE)
349  {
350  printf("\r\nFailed to register BST callback, rc = %d (%s).\r\n",
351  rc, opennsl_errmsg(rc));
352  return rc;
353  }
354  printf("BST callback for triggers is registered.\r\n");
355 
356  /* Enabling bst module. */
357  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, 1);
358  if (rc != OPENNSL_E_NONE)
359  {
360  printf("\r\nFailed to Enable bst, rc = %d (%s).\r\n",
361  rc, opennsl_errmsg(rc));
362  return rc;
363  }
364  printf("BST feature is enabled.\r\n");
365 
366 
367  while (1)
368  {
369  printf("\nUser Menu: Select one of the following options\n");
370  printf("1. Enable/Disable BST feature.\n");
371  printf("2. Display BST statistics of a port.\n");
372  printf("3. Clear BST statistics of a port.\n");
373 #ifdef INCLUDE_DIAG_SHELL
374  printf("9. Launch diagnostic shell\n");
375 #endif
376  printf("0. Quit the application.\n");
377 
378  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
379  {
380  printf("Invalid option entered. Please re-enter.\n");
381  continue;
382  }
383  switch(choice)
384  {
385  case 1:
386  {
387  printf("\r\nConfigure BST feature: 1-Enable, 0-Disable.\r\n");
388  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
389  {
390  printf("Invalid option entered. Please re-enter.\n");
391  continue;
392  }
393  if((choice != 0) && (choice != 1))
394  {
395  printf("Invalid input. Please re-enter.\n");
396  continue;
397  }
398 
399  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, choice);
400  if (rc != OPENNSL_E_NONE)
401  {
402  printf("\r\nFailed to %s BST feature, rc = %d (%s).\r\n",
403  (choice == 0)? "Disable" : "Enable", rc, opennsl_errmsg(rc));
404  return rc;
405  }
406 
407  printf("BST feature is %s.\r\n", (choice == 0)? "Disabled" : "Enabled");
408  break;
409  } /* End of case 1 */
410 
411  case 2:
412  {
413  printf("\r\nEnter the port number.\r\n");
414  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
415  {
416  printf("Invalid option entered. Please re-enter.\n");
417  continue;
418  }
419  rc = opennsl_port_gport_get (unit, port, &gport);
420  if (rc != OPENNSL_E_NONE)
421  {
422  return rc;
423  }
424  for (index = 0; index < MAX_COUNTERS; index++)
425  {
426  rc = (opennsl_cosq_bst_stat_sync(unit, id_list[index].bid));
427  if (rc != OPENNSL_E_NONE)
428  {
429  printf("\r\nFailed to sync the state of port, rc = %d (%s).\r\n",
430  rc, opennsl_errmsg(rc));
431  break;
432  }
433  }
434 
435  for (index = 0; index < MAX_COUNTERS; index++)
436  {
437  for (cosq = 0; cosq <=7; cosq++)
438  {
439  rc = opennsl_cosq_bst_stat_get(unit, gport, cosq,
440  id_list[index].bid, options, &val[index][cosq]);
441  if (rc != OPENNSL_E_NONE)
442  {
443  printf("\r\nFailed to get the port stats, rc = %d (%s).\r\n",
444  rc, opennsl_errmsg(rc));
445  break;
446  }
447  printf("BST Counter: %s for COS queue: %d is : %llu\n",
448  id_list[index].name, cosq, val[index][cosq]);
449  }
450  printf("\n");
451  }
452 
453  break;
454  } /* End of case 2 */
455 
456  case 3:
457  {
458  printf("\r\nEnter the port number.\r\n");
459  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
460  {
461  printf("Invalid option entered. Please re-enter.\n");
462  continue;
463  }
464  rc = opennsl_port_gport_get (unit, port, &gport);
465  if (rc!= OPENNSL_E_NONE)
466  {
467  return rc;
468  }
469 
470  for (index = 0; index < MAX_COUNTERS; index++)
471  {
472  for (cosq = 0; cosq <= 7; cosq++)
473  {
474  rc = opennsl_cosq_bst_stat_clear(unit, gport, cosq,
475  id_list[index].bid);
476  if (rc != OPENNSL_E_NONE)
477  {
478  printf("\r\nFailed to clear the port stats, rc = %d (%s).\r\n",
479  rc, opennsl_errmsg(rc));
480  break;
481  }
482  }
483  printf("\n");
484  }
485 
486  printf("\r\nPort %d stats cleared\r\n", port);
487  break;
488  } /* End of case 3 */
489 
490 #ifdef INCLUDE_DIAG_SHELL
491  case 9:
492  {
493  opennsl_driver_shell();
494  break;
495  }
496 #endif
497 
498  case 0:
499  {
500  printf("Exiting the application.\n");
501  return OPENNSL_E_NONE;
502  }
503 
504  default :
505  break;
506  } /* End of switch */
507  } /* End of while */
508 
509  return 0;
510 }
